# Définir une fonction de croissance exponentielle pour ajuster les données.
def exp_growth(x, a, b):
    return a * np.exp(b * x)
#Préparer les Données pour l'Ajustement de Courbe
x_data = filtered_years.index - filtered_years.index.min() #centrer les années en anée minimal
y_data = filtered_years.values
#Ajuster les Données à la Fonction de Croissance Exponentielle
params, covariance = curve_fit(exp_growth, x_data, y_data)
# Prévoir le Nombre de Véhicules Électriques pour 2024 et les Cinq Années Suivantes
forecast_years = np.arange(2024, 2024 + 6) - filtered_years.index.min()
forecasted_values = exp_growth(forecast_years, *params)
#Créer un Dictionnaire pour Afficher les Valeurs Prévisionnelles
forecasted_evs = dict(zip(forecast_years + filtered_years.index.min(), forecasted_values))
print(forecasted_evs)
