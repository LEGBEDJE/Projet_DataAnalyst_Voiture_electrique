# Dfinir une fonction de croissance exponentielle pour ajuster les donnes.
def exp_growth(x, a, b):
    return a * np.exp(b * x)
#Prparer les Donnes pour l'Ajustement de Courbe
x_data = filtered_years.index - filtered_years.index.min() #centrer les annes en ane minimal
y_data = filtered_years.values
#Ajuster les Donnes  la Fonction de Croissance Exponentielle
params, covariance = curve_fit(exp_growth, x_data, y_data)
# Prvoir le Nombre de Vhicules lectriques pour 2024 et les Cinq Annes Suivantes
forecast_years = np.arange(2024, 2024 + 6) - filtered_years.index.min()
forecasted_values = exp_growth(forecast_years, *params)
#Crer un Dictionnaire pour Afficher les Valeurs Prvisionnelles
forecasted_evs = dict(zip(forecast_years + filtered_years.index.min(), forecasted_values))
print(forecasted_evs)
